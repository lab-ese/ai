def solve_cryptarithmetic(letters, index, assigned_digits, used_digits):
    if index == len(letters):
        return is_valid_equation(assigned_digits)
        
    current_letter = letters[index]
    
    for digit in range(10):
        if digit not in used_digits:
            if digit == 0 and is_first_letter(current_letter):
                continue # First letters can't be 0
                
            assigned_digits[current_letter] = digit
            used_digits.add(digit)
            
            if solve_cryptarithmetic(letters, index + 1, assigned_digits, used_digits):
                return True
                
            # Backtrack
            del assigned_digits[current_letter]
            used_digits.remove(digit)
            
    return False
